we show that the vacuum condensate
$\Phi = \Wc\,\ee^{\ii\theta}$, decomposed into

of localized excitations under osmotic drift and diffusion;
\textbf{(III)}~wave-function collapse as
$\theta$-phase thermalization with a characteristic
timescale $\tau \sim \hb/\kB T$ in the overdamped regime;
\textbf{(IV)}~the vanishing of the QCD vacuum angle
$\bar\theta_{\rm QCD} = 0$ from the joint $\Wc$--$\theta$
potential minimum, structurally distinct from the
Peccei--Quinn mechanism;

violation vanishes above the QCD deconfinement temperature

photon polarization with the $\theta$-field

\textbf{(VII)}~the Hawking temperature and

on the horizon in classical curved spacetime.
Numerical verification scripts reproducing every figure in

\section{Result I: Heisenberg Uncertainty Principle}\label{sec:I}
%%% ═══════════════════════════════════════════════════════════

\begin{result}[Heisenberg uncertainty as Cauchy--Schwarz]
\label{thm:heisenberg}
Let $\rho(x) = \Wc(x)^2$ define a probability density and
introduce the $L^2(\rho)$ inner product
$\langle f,g\rangle_\rho \equiv \int f(x)\,g(x)\,\rho(x)\,\dd x$.
Then
\begin{equation}\label{eq:HUP}
  \boxed{\;\Delta x \cdot \Delta p \;\geq\; \frac{\hb}{2}\;}
\end{equation}
is a consequence of the Cauchy--Schwarz inequality applied to the amplitude gradient $f = \nabla\!\log\Wc$ and the shifted position coordinate $g = x - \langle x \rangle_\rho$.
\end{result}

\begin{proof}
In the Madelung representation~\cite{Madelung1927},
$\psi = \sqrt{\rho}\,\ee^{\ii S/\hb}$ maps onto the NVG condensate
$\Phi = \Wc\,\ee^{\ii\theta}$ via $\rho = \Wc^2$ and $S = \hb\theta$.
The position variance encodes the spread of the amplitude:
\begin{equation}
  (\Delta x)^2
  = \langle (x - \langle x \rangle_\rho)^2 \rangle_\rho\,.
\end{equation}
The momentum variance of the wave function $\psi = \Wc\,\ee^{\ii\theta}$ reads:
\begin{align}
  (\Delta p)^2 &= \langle \hat{p}^2 \rangle - \langle \hat{p} \rangle^2 \nonumber \\
  &= \hb^2 \left[ \langle (\nabla\log\Wc)^2 \rangle_\rho + \langle (\nabla\theta)^2 \rangle_\rho - \langle\nabla\theta\rangle_\rho^2 \right] \nonumber \\
  &\geq \hb^2 \langle (\nabla\log\Wc)^2 \rangle_\rho\,,\label{eq:dp_bound}
\end{align}
where we used $\langle\hat{p}\rangle = \hb\langle\nabla\theta\rangle_\rho$ and the fact that the phase variance $\langle(\nabla\theta)^2\rangle_\rho - \langle\nabla\theta\rangle_\rho^2 \geq 0$.

Applying the Cauchy--Schwarz inequality to $f = \nabla\!\log\Wc$ and $g = x - \langle x \rangle_\rho$ in $L^2(\rho)$ yields:
\begin{equation}\label{eq:CS}
  \langle(\nabla\!\log\Wc)^2\rangle_\rho
  \;\cdot\;
  \langle(x - \langle x \rangle_\rho)^2\rangle_\rho
  \;\geq\;
  \bigl|\langle\nabla\!\log\Wc \cdot (x - \langle x \rangle_\rho) \rangle_\rho\bigr|^2.
\end{equation}
Using $\nabla\!\log\Wc = \nabla\Wc/\Wc$ and $\rho = \Wc^2$, the cross-term simplifies via integration by parts:
\begin{equation}
  \langle\nabla\!\log\Wc \cdot (x - \langle x \rangle_\rho) \rangle_\rho
  = \int_{-\infty}^{\infty} \Wc\,\nabla\Wc \cdot (x - \langle x \rangle_\rho)\,\dd x
  = \frac{1}{2} \int_{-\infty}^{\infty} \nabla(\Wc^2) \cdot (x - \langle x \rangle_\rho)\,\dd x
  = -\frac{1}{2}\,,
\end{equation}
where the boundary terms vanish for any normalizable state ($\Wc(x) \to 0$ as $x \to \pm\infty$). Thus, the Cauchy--Schwarz inequality~\eqref{eq:CS} reduces to:
\begin{equation}\label{eq:HUP_CS_bound}
  \langle(\nabla\!\log\Wc)^2\rangle_\rho \cdot (\Delta x)^2 \;\geq\; \frac{1}{4}\,.
\end{equation}
Combining this with the momentum variance bound~\eqref{eq:dp_bound} yields:
\begin{equation}
  (\Delta p)^2 \cdot (\Delta x)^2 \;\geq\; \hb^2 \langle(\nabla\!\log\Wc)^2\rangle_\rho \cdot (\Delta x)^2 \;\geq\; \frac{\hb^2}{4}\,,
\end{equation}
which is exactly the Heisenberg uncertainty principle~\eqref{eq:HUP} upon taking the square root. Equality holds if and only if $\nabla\theta$ is constant and $\nabla\log\Wc$ is proportional to $x - \langle x \rangle_\rho$, recovering the Gaussian minimum-uncertainty wave packet.
\QED
\end{proof}

\begin{remark}
No quantization postulate is invoked.  Inequality~\eqref{eq:HUP}
is a theorem of real analysis for any $L^2$-normalizable pair
$(\Wc,\theta)$.  We have verified~\eqref{eq:HUP} numerically
for five state types: Gaussian, wide Gaussian, moving wave
packet, Schr\"odinger-cat superposition, and squeezed state,
obtaining $\Delta x\cdot\Delta p/(\hb/2) \in [1.000,\;5.3]$
(see Fig.~\ref{fig:heisenberg}).
\end{remark}

\section{Result II: Born Rule}\label{sec:II}
%%% ═══════════════════════════════════════════════════════════

\begin{result}[Born rule from stochastic carrier dynamics]
\label{thm:born}
At thermal equilibrium with a macroscopic apparatus at temperature $T$,
the spatial probability density $P(x)$ to detect a localized condensate
excitation (particle) at position $x$ is proportional to the local
condensate density:
\begin{equation}\label{eq:born}
  \boxed{\;P(x) = \frac{\Wc(x)^2}{\int \Wc(y)^2\,\dd^3y} = |\psi(x)|^2\;}
\end{equation}
where the identification follows from the stationary balance of
osmotic drift and diffusion of the excitation in the stochastic
condensate background.
\end{result}

\begin{proof}
In NVG, a particle is not a point-like entity but a localized excitation (or defect) in the vacuum condensate $\Phi(x) = \Wc(x)\,\ee^{\ii\theta(x)}$. When the system is coupled to a macroscopic measurement apparatus, the localized excitation undergoes stochastic Brownian motion due to thermal fluctuations of the background field. 

We describe the coordinate $\bm{X}(t)$ of the center of this localized excitation as a stochastic process. The probability density $P(\bm{x}, t)$ of finding the excitation at position $\bm{x}$ at time $t$ satisfies the Fokker--Planck equation:
\begin{equation}\label{eq:fokker}
  \frac{\partial P}{\partial t} + \nabla \cdot \bigl( P\,\bm{v}_{\rm drift} \bigr) \;=\; D\,\nabla^2 P\,,
\end{equation}
where $D$ is the diffusion coefficient, and $\bm{v}_{\rm drift}$ is the net drift velocity of the excitation.

In a stochastically fluctuating condensate, the drift velocity is composed of two components:
\begin{equation}\label{eq:drift}
  \bm{v}_{\rm drift} \;=\; \bm{v}_{\rm current} + \bm{v}_{\rm osmotic}\,.
\end{equation}
1. The \emph{current velocity} $\bm{v}_{\rm current}$ represents the coherent phase flow of the condensate, which in the Madelung representation is:
\begin{equation}
  \bm{v}_{\rm current} \;=\; \frac{\hb}{m}\,\nabla\theta\,.
\end{equation}
2. The \emph{osmotic velocity} $\bm{v}_{\rm osmotic}$ is a systematic drift force drawing the localized excitation along the gradient of the background field density. Since the excitation represents a local depression in the condensate (minimizing the coupling energy $-\Wc^2 V_0$), it is attracted towards regions of higher condensate density. By analogy with Nelson's stochastic mechanics~\cite{Nelson1966}, the osmotic drift velocity is proportional to the gradient of the log-density of the carrier field:
\begin{equation}\label{eq:osmotic}
  \bm{v}_{\rm osmotic} \;=\; 2D\,\nabla\log\Wc \;=\; D\,\nabla\log\Wc^2\,.
\end{equation}

At thermal equilibrium with the apparatus (rest frame of the wave packet, $\nabla\theta \approx 0$), the current velocity vanishes ($\bm{v}_{\rm current} = 0$), and the system reaches a stationary state ($\partial P/\partial t = 0$). In this limit, the probability current in Eq.~\eqref{eq:fokker} must vanish identically:
\begin{equation}\label{eq:balance}
  P\,\bm{v}_{\rm osmotic} - D\,\nabla P \;=\; 0\,.
\end{equation}
Substituting the osmotic velocity from Eq.~\eqref{eq:osmotic}:
\begin{equation}
  P \left( D\,\nabla\log\Wc^2 \right) \;=\; D\,\nabla P\,.
\end{equation}
Dividing by the diffusion coefficient $D > 0$ and dividing by $P$:
\begin{equation}
  \nabla\log P \;=\; \nabla\log\Wc^2\,.
\end{equation}
Integrating both sides yields:
\begin{equation}
  P(x) \;\propto\; \Wc(x)^2 \;=\; \rho(x)\,.
\end{equation}
Under the Madelung mapping $\psi(x) = \Wc(x)\,\ee^{\ii\theta(x)}$, normalising the probability density over the space gives the Born rule:
\begin{equation}
  P(x) \;=\; \frac{\Wc(x)^2}{\int \Wc(y)^2\,\dd^3y} \;=\; |\psi(x)|^2\,.
\end{equation}
This derivation bridges the two probability spaces (the configuration space of stochastic field fluctuations and the coordinate space of the localized excitation) through the Fokker--Planck equation, showing that the Born rule is a consequence of the balance between osmotic drift and diffusion in the thermalized vacuum condensate.
\QED
\end{proof}

\section{Result III: Wave-Function Collapse}\label{sec:III}
%%% ═══════════════════════════════════════════════════════════

\begin{result}[Collapse as $\theta$-thermalization]
\label{thm:collapse}
When the phase field $\theta$ is coupled to a thermal reservoir
(the measurement apparatus) at temperature $T$, it thermalizes
to a local minimum of $V(\theta)$ on the timescale
\begin{equation}\label{eq:tau}
  \boxed{\;\tau_{\rm collapse}
  = \frac{\hb}{\kB T}\;}
\end{equation}
At room temperature ($T = 300$~K) this gives
$\tau = 25.3$~fs.
\end{result}

\begin{proof}
We distinguish two regimes.  Near any minimum
$\theta_{\min}$ the potential is approximately quadratic,
$V \approx V_0\theta^2$, and the Caldeira--Leggett
model~\cite{CaldeiraLeggett1983} in the \emph{overdamped}
(Ohmic) regime gives the relaxation time
$\tau_{\rm loc} = \hb/(\kB T)$.
For large-angle relaxation ($\theta$ far from any minimum),
the Kramers escape problem with barrier height $V_0$ gives
a longer timescale
$\tau_{\rm Kr} \sim \tau_{\rm loc}\,
\ee^{V_0/(\kB T)}$.
However, at measurement-relevant temperatures
($\kB T \gg V_0$ for the Goldstone mode, whose bare mass
vanishes), the exponential Kramers factor is of order unity
and $\tau_{\rm Kr} \approx \tau_{\rm loc}$.
Thus, in the physically relevant regime:
\begin{equation}\label{eq:relax}
  \langle\theta(t)\rangle
  = \theta_{\min}
  + (\theta_0 - \theta_{\min})\,\ee^{-t/\tau}\,,
  \quad \tau \sim \frac{\hb}{\kB T}\,.
\end{equation}
The Born rule (Result~\ref{thm:born}) selects which minimum
is reached, with probability
$P_n \propto \ee^{-V(\theta_{\min}^{(n)})/\kB T}$.
No projection postulate is required.
\QED
\end{proof}

\begin{corollary}\label{cor:collapse}
This mechanism offers a possible dynamical resolution to the ``measurement problem'': measurement $=$
coupling $\theta$ to a macroscopic thermal bath.
Characteristic timescales:
$\tau(300\;\mathrm{K}) = 25.3$~fs (laboratory);
$\tau(10^{10}\;\mathrm{K}) \approx 7.6\times 10^{-25}$~s
(Big Bang nucleosynthesis epoch).
\end{corollary}

\section{Result IV: Strong CP Solution}\label{sec:IV}

\begin{result}[Strong CP from the condensate potential]
\label{thm:cp}
The QCD vacuum angle $\bar\theta_{\rm QCD}$ vanishes identically
at the global minimum of $V(\Wc,\theta)$:
\begin{equation}\label{eq:CP}
  \boxed{\;
  \left.\frac{\partial V}{\partial\theta}
  \right|_{\Wc = W_0,\;\theta = 0}
  \!=\; 0
  \qquad\Longrightarrow\qquad
  \bar\theta_{\rm QCD} = 0
  \;}
\end{equation}
\end{result}

\begin{proof}
The full potential reads
\begin{equation}\label{eq:potential}
  V(\Wc,\theta) = \lambda\bigl(\Wc^2 - W_0^2\bigr)^{\!2}
  - \Wc^2 V_0\cos\theta
  + \gamma_{\rm topo}\,\frac{\alpha_{\rm EM}}{8\pi}\,
    \theta\, F\tilde{F}\,.
\end{equation}
At the global minimum $(\Wc,\theta) = (W_0, 0)$:
\begin{equation}
  \frac{\partial V}{\partial\theta}\bigg|_{W_0,0}
  = W_0^2\, V_0\,\sin 0 = 0\,.
\end{equation}
The topological susceptibility
\begin{equation}
  \chi_{\rm top} = V_0\, W_0^2
  = (75.5\;\mathrm{MeV})^4
\end{equation}
agrees with lattice QCD~\cite{Borsanyi2016}.

The dynamical timescale of this CP restoration is crucial. At the QCD phase transition ($T_c \approx 157.3$~MeV), the Hubble expansion rate is $H_{\rm QCD} \approx 10^{-18}$~MeV. In contrast, the oscillation frequency of the phase $\theta$ (its effective mass) is that of the $\eta'$ meson, with which the condensate phase is identified: the Witten--Veneziano relation $m_\theta = \sqrt{6\,\chi_{\rm top}^{\rm quenched}}/f_\pi \approx 967$~MeV reproduces $m_{\eta'} = 958$~MeV (here $\chi_{\rm top}^{\rm quenched} \approx (191~\text{MeV})^4$ and $f_\pi = 92.4$~MeV). This gives the number of oscillations per Hubble time as $N_{\rm osc} = m_\theta/H_{\rm QCD} \approx 10^{18} \gg 1$. Thus, the phase $\theta$ relaxes to 0 dynamically and instantaneously compared to the cosmological expansion timescale, leaving no out-of-equilibrium CP-violating relics.
\QED
\end{proof}

\begin{remark}[Comparison with Peccei--Quinn]
Both PQ and NVG share the same mathematical structure:
a cosine potential whose minimum forces $\theta = 0$.
The distinction is physical.  In PQ, an \emph{ad hoc}
global $U(1)_{\rm PQ}$ symmetry is postulated, and the
axion is its pseudo-Goldstone boson — a new particle with
specific mass and couplings that must be discovered
experimentally.  In NVG, the cosine term arises from the
\emph{already-present} QCD anomaly acting on the existing
condensate $\Wc$; no new symmetry and no new particle are
introduced.  A second difference: PQ requires
$f_a \gg W_0$ (typically $\sim 10^{12}$~GeV) to avoid
astrophysical constraints, whereas NVG operates at
$W_0 = 859$~MeV.  The two mechanisms are therefore
experimentally distinguishable: if the axion is found,
PQ is confirmed and NVG's CP solution is falsified.
\end{remark}

In NVG, $\bar\theta_{\rm QCD} = 0$ is not fine-tuned but a structural consequence of potential minimization. The identification of the Goldstone phase $\theta$ of the vacuum condensate with the physical QCD vacuum angle $\bar\theta_{\rm QCD}$ is a structural postulate of the framework: the axial anomaly couples the topological charge density $G\tilde{G}$ to the chiral phase of the quark mass matrix, which in NVG is mapped to the phase of the macroscopic condensate $\Phi$.
\end{remark}

\begin{proof}
The key distinction from the standard derivation is that
the $\theta$-field is a \emph{material} degree of freedom

calculation~\cite{Hawking1975} requires quantization of a
free field in the Schwarzschild background; here we use
only the classical surface gravity and the classical
thermalization of $\theta$.

At the event horizon $r = r_s$, the surface gravity is
$\kappa = c^2/(2r_s)$.  The characteristic timescale for
$\theta$-oscillations at the horizon is set by $\kappa$:
\begin{equation}
  \tau_\theta
  = \frac{2\pi c}{\kappa}
  = \frac{4\pi r_s}{c}
  = \frac{8\pi G M}{c^3}\,.
\end{equation}
Applying Result~\ref{thm:collapse}
($T = \hb/\kB\tau_\theta$) to the $\theta$-mode at the
horizon:
\begin{equation}
  T = \frac{\hb}{\kB\,\tau_\theta}
  = \frac{\hb\, c^3}{8\pi\, G\, M\, \kB}
  \;\equiv\; T_H\,.
\end{equation}
The numerical coincidence with the Hawking
temperature is non-trivial: the standard derivation uses
Bogoliubov coefficients of a quantized field, while NVG
uses the thermalization of a classical condensate mode.
That both give the same $T_H$ is a consistency check.

\emph{Caveat on the reservoir assumption.}
Result~\ref{thm:collapse} was derived for a $\theta$-mode
coupled to an external thermal reservoir at fixed $T$.
At the black hole horizon, no external reservoir is present;
the temperature is \emph{itself} a consequence of the
horizon geometry.  The application of Result~\ref{thm:collapse}
here is therefore not a direct logical deduction but an
extension by analogy: we identify the surface-gravity
timescale $\tau_\theta = 2\pi c/\kappa$ with the
thermalization time of Result~\ref{thm:collapse} and
\emph{define} the horizon temperature accordingly.
The self-consistency of this identification (correct $T_H$,
correct $S_{\rm BH}$) supports but does not rigorously
prove the extension.  A first-principles derivation of
$\theta$-thermalization on the horizon from the NVG
action~\eqref{eq:action} in a Schwarzschild background
remains an open problem.

For the entropy: each independent $\theta$-mode occupies one

$N_\theta = A/(4\ell_{\rm Pl}^2) = S_{\rm BH}/\kB$.

\QED
\end{proof}

However, its physical relevance in NVG relies on identifying the vacuum condensate amplitude $\Wc^2$ with the probability density of a particle, which leads to Result II.
  
  \item \textbf{Result II (Born Rule):} The derivation of the Born rule via the Fokker--Planck equation connects the coordinate probability density $P(x)$ to the field density $\Wc(x)^2$ by balancing the osmotic drift and diffusion of a localized defect. This provides a rigorous bridge between the field configuration space and the coordinate space. However, it relies on the physical postulate that the defect's diffusion coefficient is exactly $D = \hb/(2m)$, and that the defect couples to the background field gradient through the logarithmic osmotic velocity $\bm{v}_{\rm osmotic} = D \nabla \log \Wc^2$. A first-principles derivation of these parameters from the underlying NVG Lagrangian remains an open problem, placing this result in the semi-rigorous category (see also Nelson~\cite{Nelson1966}).
  
  \item \textbf{Result III (Wave-Function Collapse):} Although the Caldeira--Leggett model provides a robust dynamical mechanism for phase relaxation ($\tau \propto \hb/\kB T$) when coupled to a macroscopic heat bath (measurement apparatus), it remains an open question how the environment selects a particular localized measurement pointer state in a general multi-body system without a pre-defined division of the system and apparatus.
  
  \item \textbf{Result IV (Strong CP Solution):} Minimization of the vacuum potential $V(\Wc, \theta)$ naturally forces the Goldstone phase $\theta$ to zero. However, this relies on the physical postulate that the physical QCD vacuum angle $\bar\theta_{\rm QCD}$ is identical to the phase of the macroscopic NVG condensate $\theta$. If this identification is broken by higher-order gravitational corrections, a small residual $\bar\theta_{\rm QCD}$ could emerge.
  
  \item \textbf{Result V (Arrow of Time):} The Arrow of Time is established by selecting the topological sector $Q = +1$ at the cosmological bounce, which determines the direction of coordinates in which $\dot\theta > 0$. While this provides a consistent topological interpretation, it does not prevent micro-reversibility at the underlying equations of motion, representing an interpretational mapping rather than an absolute mathematical proof of asymmetry.
  
  \item \textbf{Result VI (Bell--CHSH Violation):} The temperature-dependent CHSH parameter relies on the postulate that photon polarization correlations in the condensate background behave as $E(\bm{a},\bm{b}) = -\cos 2(\theta_a - \theta_b)$. A first-principles derivation of this correlation function from the photon propagator in the stochastically fluctuating condensate remains an open challenge.
  
  \item \textbf{Result VII (Quantum Gravity):} The Hawking temperature is obtained by identifying the surface gravity timescale $\tau_\theta = 2\pi c/\kappa$ with the thermalization time of the stochastically coupled Goldstone phase. This extension of the reservoir model to a black hole horizon is heuristic and relies on the classical curved spacetime approximation. A full quantum-field-theoretic treatment of the $\theta$-mode in a black hole background is needed to verify this equivalence.
\end{itemize}

\subsection{Bulk Matter and Radiation Entropy}
The bulk thermodynamic entropy of matter and radiation, $S_{\rm obs}$, is generated at each bounce via nonperturbative decay of the vacuum condensate (analogous to reheating) driven by the topological winding $Q = \frac{1}{2\pi}\oint \dd\theta = +1$ of the Goldstone phase.

In the first cycle (Genesis), the universe is born with a hot QGP core of volume $V_1 = \frac{4}{3}\pi r_c^3 \approx 5.99 \times 10^{15} \text{ cm}^3$ at the bounce temperature $T_b \approx 432$ MeV. The initial bulk thermodynamic entropy of this relativistic gas is:
\begin{equation}
  S_{\rm obs, 1} = s(T_b)\,V_1 = \frac{2\pi^2}{45} g_{*S} \left( \frac{\kB T_b}{\hbar c} \right)^3 V_1 \approx 1.31 \times 10^{57} \, k_B,
\end{equation}
where $g_{*S} \approx 47.5$ is the effective number of relativistic degrees of freedom for a three-flavor quark-gluon plasma.

The present-day bulk entropy does not require the inter-cycle ladder at all:
it follows adiabatically from the most recent bounce. Entropy conservation
$s(T)\,a^3 = \mathrm{const}$ between the cycle-77 bounce ($T_b \approx 432$~MeV)
and today ($T_0 = 2.7255$~K) fixes the present entropy density at the standard
value $s_0 \approx 2.9 \times 10^{3}\,k_B\,\mathrm{cm}^{-3}$ (photons and
neutrinos), so that the entropy within the present Hubble volume is
\begin{equation}
  S_{\rm obs} \,=\, s_0 \cdot \frac{4}{3}\pi R_{H_0}^{3}
  \,\approx\, 2 \times 10^{88}\,k_B,
\end{equation}
in agreement with the standard evaluation
($S_{\rm obs} \approx 1.7\text{--}2.5 \times 10^{88}\,k_B$ depending on the
volume convention). We emphasize the honest status of this statement: it is a
\emph{consistency} of the bounce temperature with standard adiabatic
expansion, not an independent prediction, and it carries no free parameter.
(An earlier version of this appendix reproduced $10^{88}$ through a compounded
per-cycle factor $f_S \approx 2.545$; that factor was fitted to the observed
entropy and is withdrawn. It also used $R_n \propto \sqrt{M_n}$ for the
radiation era, which is dynamically inconsistent with the turnaround relation
$R_n \propto M_n$ used above; the exact radiation-era exponent is
$R_n \propto M_n^{2/3}$, while the late, matter-dominated cycles obey
$R_n \propto M_n$ — the regime in which the doubling chain of the previous
subsection is valid.)

The bulk entropy remains far below the holographic horizon bound
$S_{\rm hor, 77} \approx 8.8 \times 10^{121}\,k_B$:
\begin{equation}
  \frac{S_{\rm obs}}{S_{\rm hor, 77}} \approx
  \frac{2 \times 10^{88}}{8.8 \times 10^{121}} \approx 2 \times 10^{-34},
\end{equation}
satisfying the holographic principle at all epochs.
